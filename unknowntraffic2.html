üêç Step 4: Complete Solution Script

I developed a comprehensive Python script to extract the ICMP chunks, reassemble them, decode the image, and scan the QR code:

from scapy.all import rdpcap, ICMP, IP, Raw
import re, base64, cv2
import numpy as np

def estrai_icmp_chunks(pcap_file, src_ip="10.0.0.10"):
    # Read packets from the input pcap file
    pacchetti = rdpcap(pcap_file)
    chunks = []
    
    for pkt in pacchetti:
        # Check if the packet is IP + ICMP, from the given source, and has raw payload
        if IP in pkt and ICMP in pkt and pkt[IP].src == src_ip and Raw in pkt:
            try:
                # Decode the raw payload as string (ignore errors for non-text bytes)
                payload = pkt[Raw].load.decode(errors="ignore")
                
                # Match payloads starting with CHUNK_### (ordinal number)
                match = re.match(r"CHUNK_(\d{3})(.*)", payload, re.DOTALL)
                if match:
                    num = int(match.group(1))           # Extract the chunk number
                    contenuto = match.group(2).strip()  # Extract the remaining content
                    chunks.append((num, contenuto))     # Store as tuple (num, content)
            except Exception:
                # Skip packet if decoding/parsing fails
                continue
    
    # Sort the chunks by their numeric order (000, 001, 002, ...)
    chunks.sort(key=lambda x: x[0])
    
    # Return only the ordered contents (without numbers)
    return [c[1] for c in chunks]

if __name__ == "__main__":
    input_pcap = "traffic.pcap"
    
    # Extract and order all chunks from ICMP packets
    array_dati = estrai_icmp_chunks(input_pcap)
    
    # Concatenate all chunks and decode the base64 image data
    nparr = np.frombuffer(base64.b64decode(''.join(array_dati)), np.uint8)
    img = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
    
    # Use OpenCV to detect and decode the QR code
    detector = cv2.QRCodeDetector()
    data, bbox, _ = detector.detectAndDecode(img)
    
    print("Flag:", data)

üß† How this works:

  1. Extract all ICMP packets from the specified source IP (10.0.0.10)
  2. Parse each payload to find chunks with the CHUNK_### pattern
  3. Sort chunks by their numerical order
  4. Concatenate all Base64 data and decode to get image bytes
  5. Use OpenCV to decode the image and scan for QR codes
  6. Extract and display the flag from the QR code
